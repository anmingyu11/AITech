i=1

for name_1 in cont_names:
    j=cont_names.index(name_1)

    while(j<len(cont_names)-1):
        plt.subplot(6,1,i)
        plt.title(name_1 + ' vs ' + cont_names[j+1])
        sns.jointplot(x=name_1,y=cont_names[j+1],data=train_df) 
        j=j+1
        i=i+1
